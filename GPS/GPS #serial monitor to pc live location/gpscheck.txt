#include <WiFi.h>
#include <HardwareSerial.h>
#include <TinyGPSPlus.h>
#include <ESPAsyncWebServer.h>

const char* ssid = "Prithviraj";       // ðŸ” Replace with your hotspot SSID
const char* password = "PrithviraJK123";   // ðŸ” Replace with your hotspot password

// Initialize GPS parser and Serial2 for GPS module
TinyGPSPlus gps;
HardwareSerial gpsSerial(1); // Use UART1 on ESP32

AsyncWebServer server(80);

void setup() {
  Serial.begin(115200);
  gpsSerial.begin(9600, SERIAL_8N1, 33, 32); // RX=33, TX=32

  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nâœ… Connected to WiFi");
  Serial.print("ESP32 IP Address: ");
  Serial.println(WiFi.localIP());

  server.on("/", HTTP_GET, [](AsyncWebServerRequest *request){
    request->send(200, "text/plain", "ESP32 GPS Server is running");
  });

  server.begin();
}

void loop() {
  // Parse incoming GPS data
  while (gpsSerial.available()) {
    gps.encode(gpsSerial.read());
  }

  // If valid GPS location is available, print it
  if (gps.location.isUpdated()) {
    Serial.println("ðŸ“¡ GPS Data:");
    Serial.print("Latitude: ");
    Serial.println(gps.location.lat(), 6);
    Serial.print("Longitude: ");
    Serial.println(gps.location.lng(), 6);
    Serial.println("-----------");
  }

  delay(1000);
}
